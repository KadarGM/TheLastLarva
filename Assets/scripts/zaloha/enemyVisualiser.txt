func _process(_delta: float) -> void:
	if Engine.is_editor_hint():
		if enemy.enemy_data.body:
			_set_sprites()
			_set_colors()
		else:
			pass

func _set_sprites() -> void:
	body.texture = enemy.enemy_data.body
	sword_body.texture = enemy.enemy_data.sword
	sword_body_2.texture = enemy.enemy_data.sword
	leg_f.texture = enemy.enemy_data.leg
	feet_f.texture = enemy.enemy_data.feet
	leg_b.texture = enemy.enemy_data.leg
	feet_b.texture = enemy.enemy_data.feet
	arm_f_1.texture = enemy.enemy_data.arm
	hand_f_1.texture = enemy.enemy_data.hand

	arm_b_1.texture = enemy.enemy_data.arm
	hand_b_1.texture = enemy.enemy_data.hand
	sword_b.texture = enemy.enemy_data.sword
	arm_f_2.texture = enemy.enemy_data.arm
	hand_f_2.texture = enemy.enemy_data.hand
	arm_b_2.texture = enemy.enemy_data.arm
	hand_b_2.texture = enemy.enemy_data.hand
	head.texture = enemy.enemy_data.head
	eye_b.texture = enemy.enemy_data.eye_b
	eye_f.texture = enemy.enemy_data.eye_f
	feller_b_1.texture = enemy.enemy_data.feller_1
	feller_b_2.texture = enemy.enemy_data.feller_2

	mandible_f.self_modulate = enemy.enemy_data.mandibles_color
	mandible_b.self_modulate = enemy.enemy_data.mandibles_color

	sword_body_2.self_modulate = enemy.enemy_data.weapon_color
